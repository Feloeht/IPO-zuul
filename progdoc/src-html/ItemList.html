001import java.util.HashMap;
002import java.util.Set;
003/**
004 * Cete classe fait parti du jeu "Le colis perdu"
005 * Cette classe permet la création et la gestion de l'inventaire du joueur
006 * 
007 * @author  Théo LEFEVRE
008 * @version 2019.05.08
009 */
010public class ItemList
011{
012    private HashMap<String, Item> aItemList;
013    /**
014     * Constructeur d'objets de classe ItemList
015     */
016    public ItemList()
017    {
018        this.aItemList = new HashMap<String, Item>();
019    } // ItemList()
020    
021    /**
022     * Procedure ajout item dans hashmap
023     * @param   pNom    Nom de l'item sous forme de string
024     * @param   pItem   Item associé
025     */
026    public void addItem(final String pNom, final Item pItem){
027        this.aItemList.put(pNom, pItem);
028    } // addItem()
029    
030    /**
031     * Procedure retrait item dans hashmap
032     * @param   pNom    Nom de l'item sous forme de string
033     */
034    public void removeItem(final String pNom){
035        this.aItemList.remove(pNom);
036    } // removeItem()
037    
038    /**
039     * Methode retour Item (adresse) associé a une string
040     * @param pItem nom de l'item sous forme de string
041     * @return Item
042     */
043    public Item getItem(final String pItem){
044        return this.aItemList.get(pItem);
045    } // getItem()
046    
047    /**
048     * Methode test presence Item dans inventaire joueur
049     * @param pNom Nom de l'Item a chercher
050     * @return Boolean true ou false si vide ou non
051     */
052    public boolean hasItem(final String pNom){
053        return aItemList.containsKey(pNom);
054    } // hasItem()
055    
056    /**
057     * Methode test inventaire vide
058     * @return true ou false si vide ou non
059     */
060    public boolean isEmpty(){
061        return aItemList.isEmpty();
062    } // isEmpty()
063    
064    /**
065     * Methode retour Item présents dans une Room
066     * @return  Item de la Room
067     */
068    public String getItemString(){
069        String vItems = "";
070        Set<String> vKeys = this.aItemList.keySet();
071        for(String item : vKeys)
072        {
073            vItems += " " + item;
074        }
075        return vItems;
076    } // getItemString()
077}
