123    /**
124     * Given a command, process (that is: execute) the command.
125     * If this command ends the game, true is returned, otherwise false is
126     * returned.
127     * @param pCommande la commande tapee
128     */
129    public void interpretCommand(final String pCommande) 
130    {
131        gui.println(pCommande);
132        Command vCommande = parser.getCommand(pCommande);
133        if(vCommande.isUnknown()) {
134            this.gui.println("Commande inconnue... \n");
135            return;
136        }
137        String vCommandWord = vCommande.getCommandWord();
138        if(vCommandWord.equals("go"))this.goRoom(vCommande);
139        else if(vCommandWord.equals("look"))this.look(vCommande);
140        else if(vCommandWord.equals("take"))this.take(vCommande);
141        else if(vCommandWord.equals("drop"))this.drop(vCommande);
142        else if(vCommandWord.equals("eat"))this.eat(vCommande);
143        else if(vCommandWord.equals("test"))this.test(vCommande);
144        else if(vCommandWord.equals("charge"))this.charge(vCommande);
145        else if(vCommandWord.equals("use"))this.use(vCommande);
146        else if(vCommandWord.equals("inventaire")){
147            if(vCommande.hasSecondWord())this.gui.println("Commande sans paramêtre \n");
148            else this.gui.println(this.aPlayer.inventaire());
149        }
150        else if(vCommandWord.equals("help")){
151            if(vCommande.hasSecondWord())this.gui.println("Commande sans paramêtre \n");
152            else this.printHelp();
153        }
154        else if (vCommandWord.equals("quit")) {
155            if(vCommande.hasSecondWord())this.gui.println("Commande sans paramêtre \n");
156            else this.endGame();
157        }
158        else if (vCommandWord.equals("back")) {
159            if(vCommande.hasSecondWord())this.gui.println("Commande sans paramêtre \n");
160            else this.back();
161        }
162    } // interpretCommand()

164    /** 
165     * Try to go to one direction. If there is an exit, enter the new
166     * room, otherwise print an error message.
167     * @param pDirection   Direction vers laquelle on veut aller
168     */
169    private void goRoom(final Command pDirection) 
170    {
171        if(!pDirection.hasSecondWord()) {
172            this.gui.println("Aller où ? \n");
173            return;
174        }
175        String vDirection = pDirection.getSecondWord();
176        Room vNextRoom = this.aPlayer.getPosition().getExit(vDirection);
177        if (vNextRoom == null)
178            this.gui.println("Cette sortie n'existe pas \n");
179        else {
180            this.aDeplacements += 1;
181            this.aPlayer.setBackRoom(this.aPlayer.getPosition());
182            this.aPlayer.setPosition(vNextRoom);
183            if(this.aDeplacements == 10){
184                this.gui.println("Vous n'avez pas mangé durant votre aventure.");
185                this.gui.println("Vous êtes mort de faim.");
186                this.endGame();
187            }
188            else{
189                this.gui.println(this.aPlayer.getPosition().getLongDescription());
190            }
191            if(this.aPlayer.getPosition().getImageName() != null) this.gui.showImage(this.aPlayer.getPosition().getImageName());
192        }
193    } // goRoom(

215    /**
216     * procédure pour manger magic cookie
217     * @param pCommand Item a manger
218     */
219    public void eat(final Command pCommand)
220    {
221        String vNom = pCommand.getSecondWord();
222        Item vItem = this.aPlayer.getInventory().getItem(vNom);
223        if(!pCommand.hasSecondWord())
224        {
225            this.gui.println("Manger quoi ? \n");
226            return;
227        }
228        else if(!this.aPlayer.getInventory().hasItem(vNom)){
229            this.gui.println("Cet item n'est pas dans votre inventaire. \n");
230            return;
231        }
232        else if(vNom.equals("cookie")){
233            this.aPlayer.getInventory().removeItem(vNom);
234            this.aPlayer.setPoidsMax(this.aPlayer.getPoidsMax() * 2);
235            this.gui.println("Vous avez mangé un cookie magique, vous n'avez plus faim !");
236            this.gui.println("La charge maximale que vous pouvez porter a été doublée !\n");
237        }
238        else this.gui.println("Cet Item n'est pas mangeable, ou vous n'avez plus faim. \n");
239    } // eat()

241    /**
242     * Procedure retour arriere
243     */
244    private void back()
245    {
246        if(!this.aPlayer.backRoomIsEmpty()){
247            if(this.aPlayer.getPosition().getTrapedDoor()){
248                this.aPlayer.clearBackRoom();
249                this.gui.println("Cette porte est verrouillée, impossible de retourner en arrière.");
250            }
251            else{
252                this.aPlayer.setPosition(this.aPlayer.getBackRoom());
253                this.gui.println(this.aPlayer.getPosition().getLongDescription());
254                if(this.aPlayer.getPosition().getImageName() != null){
255                    gui.showImage(this.aPlayer.getPosition().getImageName());
256                }
257            }
258        }
259        else{this.gui.println("Vous êtes déjà à votre point de départ");}
260    } // back()

262    /**
263     * Procedure prendre item
264     * @param pCommand  item a prendre dans la room
265     */
266    private void take(final Command pCommand){
267        String vNom = pCommand.getSecondWord();
268        Item vItem = this.aPlayer.getPosition().getItemList().getItem(vNom);
269        if(!pCommand.hasSecondWord()){
270            this.gui.println("Prendre quoi ? \n");
271            return;
272        }
273        else if(!this.aPlayer.getPosition().getItemList().hasItem(vNom)){
274            this.gui.println("Cet item n'existe pas. \n");
275            return;
276        }
277        else if(!this.aPlayer.takePoidsOK(vItem.getPoidsItem())){
278            this.gui.println("Objet trop lourd ! \n");
279        }
280        else{
281            this.aPlayer.getInventory().addItem(vNom, vItem);
282            this.aPlayer.getPosition().getItemList().removeItem(vNom);
283            this.gui.println("Vous avez ramassé " + this.aPlayer.getPlayerItem(vNom));
284            this.gui.println(this.aPlayer.inventaire() + "\n");
285        }
286    } // take()

332    /**
333     * Procedure utilisation beamer
334     * @param pCommand Beamer de l'inventaire à utiliser
335     */
336    private void use(final Command pCommand){
337        String vNom = pCommand.getSecondWord();
338        Beamer vBeamer = (Beamer)this.aPlayer.getInventory().getItem(vNom);
339        if(!pCommand.hasSecondWord()){
340            this.gui.println("Utiliser quoi ? \n");
341            return;
342        }
343        else if(!this.aPlayer.getInventory().hasItem(vNom)){
344            this.gui.println("Vous ne possédez pas ce Beamer. \n");
345            return;
346        }
347        else if(!vBeamer.isCharged()){
348            this.gui.println("Ce Beamer n'est pas chargé. \n");
349            return;
350        }
351        else if(this.aPlayer.getPosition() == vBeamer.getChargeRoom()){
352            this.gui.println("Vous êtes déjà dans la salle chargée par le Beamer. \n");
353            return;
354        }
355        else{
356            this.aPlayer.clearBackRoom();
357            this.aPlayer.setPosition(vBeamer.getChargeRoom());
358            vBeamer.dechargeBeamer();
359            this.gui.println("Vous avez utilisé ce Beamer ! \n");
360            this.gui.println(this.aPlayer.getPosition().getLongDescription());
361            if(this.aPlayer.getPosition().getImageName() != null){
362                gui.showImage(this.aPlayer.getPosition().getImageName());
363            }
364        }
365    } // use()

367    /**
368     * Procedure test
369     * @param pCommand fichier de test sans .txt
370     */
371    private void test(final Command pCommand)
372    {
373        if(!pCommand.hasSecondWord())
374        {
375            this.gui.println("Tester quoi ? \n");
376            return;
377        }      
378        String vFichier = pCommand.getSecondWord();
379        Scanner vScanner = null;
380        try{
381            vScanner = new Scanner(new File("tests/" + vFichier + ".txt"));
382        }
383        catch(java.io.FileNotFoundException pE){
384            System.out.println( "Erreur -> " + pE.getMessage() );
385        }
386        while(vScanner.hasNextLine())
387        {
388            String vCommande = vScanner.nextLine();
389            this.interpretCommand(vCommande);
390        }
391        if(vScanner != null) vScanner.close();
392    } // test()
