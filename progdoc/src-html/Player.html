001import java.util.Stack;
002/**
003 * Cete classe fait parti du jeu "Le colis perdu"
004 * Cette classe permet la création des joueurs du jeu
005 * et de fonctions leur étant associées
006 * 
007 * @author  Théo LEFEVRE
008 * @version 2019.05.08
009 */
010public class Player
011{
012    private String aNom;
013    private double aPoidsJoueur;
014    private double aPoidsMax;
015    private Room aCurrentRoom;
016    private Stack<Room> aBackRoom;
017    private ItemList aItemList;
018    
019    /**
020     * Constructeur d'objets de classe Player
021     * @param pNom Nom du joueur
022     */
023    public Player(final String pNom)
024    {
025        this.aNom = pNom;
026        this.aPoidsJoueur = 60; 
027        this.aPoidsMax = 90;
028        this.aBackRoom = new Stack<Room>();
029        this.aItemList = new ItemList();
030    } // Player()
031
032    /**
033     * Accesseur nom du joueur
034     * @return Nom du joueur
035     */
036    public String getNom(){
037        return this.aNom;
038    } // getNom()
039    
040    /**
041     * Accesseur inventaire du joueur
042     * @return Inventaire du joueur
043     */
044    public ItemList getInventory(){
045        return this.aItemList;
046    } // getItemList()
047    
048    /**
049     * Accesseur position du joueur
050     * @return Localisation actuelle du player
051     */
052    public Room getPosition(){
053        return this.aCurrentRoom;
054    } // getPosition()
055
056    /**
057     * Accesseur salle precedente
058     * @return Salle precedente
059     */
060    public Room getBackRoom(){
061        return this.aBackRoom.pop();
062    } // getBackRoom()
063    
064    /**
065     * Accesseur poids que peut porter le joueur
066     * @return Poids que peut porter le joueur
067     */
068    public double getPoids(){
069        return this.aPoidsJoueur;
070    } // getPoids()
071        
072    /**
073     * Accesseur poids max que peut porter le joueur
074     * @return Poids max
075     */
076    public double getPoidsMax(){
077        return this.aPoidsMax;
078    } // getPoidsMax()
079    
080    /**
081     * Modificateur poids que peut porter le joueur
082     * @param pPoids Poids que peut porter le joueur
083     */
084    public void setPoids(final double pPoids){
085        this.aPoidsJoueur = pPoids;
086    } // setPoids()
087    
088    /**
089     * Modificateur poids max que peut porter le joueur
090     * @param pPoidsMax Poids max
091     */
092    public void setPoidsMax(final double pPoidsMax){
093        this.aPoidsJoueur = pPoidsMax;
094    } // setPoidsMax()
095    
096    /**
097     * Methode test si le joueur depasse sa capacite de charge
098     * @param pPoidsItem Le Poids d'un item a tester
099     * @return true ou false si depasse ou pas
100     */
101    public boolean takePoidsOK(final double pPoidsItem){
102        return this.aPoidsJoueur + pPoidsItem <= this.aPoidsMax;
103    } // takePoidsOK()
104    
105    /**
106     * Modificateur de la position du joueur
107     * @param pPosition Room dans laquelle le joueur se trouvera
108     */
109    public void setPosition(final Room pPosition){
110        this.aCurrentRoom = pPosition;
111    } // setPosition()
112    
113    /**
114     * Procedure ajout Room dans la pile de backroom
115     * @param pCurrentRoom Rom actuelle du joueur
116     */
117    public void setBackRoom(final Room pCurrentRoom){
118        this.aBackRoom.push(pCurrentRoom);
119    } // setBackRoom()
120    
121    /**
122     * Procedure vidage pile de backroom
123     */
124    public void clearBackRoom(){
125        this.aBackRoom.clear();
126    } // setBackRoom()
127    
128    /**
129     * Methode test si Stack vide
130     * @return true ou false si vide ou non
131     */
132    public boolean backRoomIsEmpty(){
133        return this.aBackRoom.empty();
134    } // backRoomIsEmpty
135    
136    /**
137     * Methode retour Item (adresse) associé a une string
138     * @param pNom nom de l'item sous forme de string
139     * @return Item
140     */
141    public Item getPlayerItem(final String pNom){
142        return this.aItemList.getItem(pNom);
143    } // getPlayerItem()
144    
145    /**
146     * Methode affichage de l'inventaire
147     * @return inventaire sous forme de string
148     */
149    public String inventaire(){
150        if(this.aItemList.isEmpty()) return "Votre inventaire est vide";
151        else return "Vous possedez : " + this.aItemList.getItemString();
152    } // inventaire()
153    
154} // Player
