001import java.util.HashMap;
002import java.util.Set;
003import java.util.Iterator;
004/**
005 * Cete classe fait parti du jeu "Le colis perdu"
006 * Cette classe permet la création des salles du jeu 
007 * et la gestion de leur contenu
008 * 
009 * @author  Théo LEFEVRE
010 * @version 2019.05.08
011 */
012public class Room
013{
014    private String aDescription;
015    private String imageName;
016    private ItemList aItemList;
017    private HashMap<String, Room> exits;
018    private boolean aTrapDoor;
019
020    /**
021     * Item constructor
022     * @param pDescription  Description de la Room
023     * @param pImage        Image de la Room
024     * @param pTraped       Porte piege
025     */
026    public Room(final String pDescription, final String pImage, final boolean pTraped){
027        this.aDescription = pDescription;
028        this.imageName = pImage;
029        this.aItemList = new ItemList();
030        this.exits = new HashMap<String, Room>();
031        this.aTrapDoor = pTraped;
032    } // Room()
033    
034    /**
035     * Accesseur description Room
036     * @return Description de la salle
037     */
038    public String getDescription(){
039        return this.aDescription;
040    } // getDescription()
041
042    /**
043     * Accesseur TrapDoor sous forme de boolean
044     * @return true ou false si porte piege ou non
045     */
046    public boolean getTrapedDoor(){
047        return aTrapDoor;
048    } // getTrapedDoor()
049    
050    /**
051     * Accesseur ItemList
052     * @return Liste des Items sous forme ItemList
053     */
054    public ItemList getItemList(){
055        return this.aItemList;
056    } // getItemList()
057    
058    /**
059     * Procedure denition exits des Room
060     * @param pDirection    La direction possible
061     * @param pRoom         La Room vers laquelle elle pointe
062     */
063    public void setExits(final String pDirection, final Room pRoom)
064    {
065        exits.put(pDirection, pRoom);
066    } // setExits()
067
068    /**
069     * Methode affichage direction exits
070     * @param pDirection    Direction
071     * @return Salle associée a la direction
072     */
073    public Room getExit(final String pDirection)
074    {
075        return exits.get(pDirection);
076    } // getExit()
077
078    /**
079     * Methode affichage des sorties
080     * @return Sorties de la salle actuelle, sous forme d'une chaine de String
081     */
082    public String getExitString()
083    {
084        String vExits = "Les sorties :";
085        Set<String> keys = exits.keySet();
086        for(String exit : keys) {
087            vExits += " " + exit;
088        }
089        return vExits;
090    } // getExitString()
091    
092    /**
093     * Methode affichage avec test des sorties destine aux String des Rooms
094     * @return La liste des Items presents dans une Room
095     */
096    private String getItemString(){
097        if(this.getItemList().isEmpty()){
098            return "Aucun objet dans cette salle. \n";
099        }
100        else{
101            return "Les Items : " + this.aItemList.getItemString() + "\n";
102        }
103    } // getItemString()
104    
105    /**
106     * Methode affichage description complete Room
107     * @return Description complète de la salle actuelle
108     */
109    public String getLongDescription()
110    {
111        return "Vous etes " + this.getDescription() + "\n" 
112                            + this.getExitString() + "\n" 
113                            + this.getItemString();
114    } // getLongDescription()
115    
116    /**
117     * Return a string describing the room's image name
118     * @return nom de l'image
119     */
120    public String getImageName()
121    {
122        return this.imageName;
123    } // getImageName()
124    
125} // Room
